Goal: Get location map image for a resource.

Request: GET /api/v1/resources/{id}/location

import requests
import base64

BASE_URL = "https://lib-btaageoapi-dev-app-01.oit.umn.edu/api/v1"

# Get location map image for a resource
resource_id = "ark28722-s7vs38"
response = requests.get(f"{BASE_URL}/resources/{resource_id}/location")

if response.status_code == 200:
    # Save the image
    with open('location_map.png', 'wb') as f:
        f.write(response.content)
    print(f"‚úÖ Location map image saved for resource: {resource_id}")
    print(f"Content-Type: {response.headers.get('Content-Type', 'N/A')}")
    print(f"Image size: {len(response.content)} bytes")
else:
    print(f"‚ùå Error: {response.status_code}")
